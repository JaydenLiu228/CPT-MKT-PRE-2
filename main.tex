\section{Introduction}

\begin{frame}{Motivation}
    \begin{itemize}
        \item \textbf{Regulatory Emphasis:} The SEC \citeyearpar{sec2000} stresses that comparable financial statements facilitate \textbf{efficient capital allocation} and nurture investor confidence.
        \item \textbf{FASB Conceptual Framework:} ``Investing and lending decisions \dots\ cannot be made rationally if comparative information is not available'' \citep{fasb1980}.
        \item \textbf{Practical Importance:} Financial analysis textbooks universally emphasize cross-firm comparability for ratio analysis and performance benchmarking.
        \item \textbf{The Gap:} Despite widespread agreement on its importance, \textcolor{hkustred}{\textbf{no empirical construct}} of comparability existed, and \textcolor{hkustred}{\textbf{little evidence}} documented its benefits to users.
    \end{itemize}
\end{frame}

\begin{frame}{Research Question}
    \vspace{0.5cm}
    \begin{center}
        \Large \textit{``How can we \textbf{measure} financial statement comparability, and does higher comparability \textbf{benefit} financial statement users?''}
    \end{center}
    \vspace{0.5cm}
    \begin{itemize}
        \item \textbf{Develop} a firm-specific, output-based, quantitative measure of financial statement comparability.
        \item \textbf{Document} benefits to sell-side analysts: coverage, forecast accuracy, and forecast dispersion.
    \end{itemize}
\end{frame}

% Theoretical Foundation

\section{Theoretical Foundation}

\begin{frame}{Conceptual Definition of Comparability}
    \begin{block}{FASB Definition}
        ``Comparability is the quality of information that enables users to identify \textbf{similarities and differences} between two sets of economic phenomena'' \citep{fasb1980}.
    \end{block}
    \vspace{0.3cm}
    \begin{itemize}
        \item The \textbf{accounting system} is a mapping from economic events to financial statements:
        \[
            \text{Financial Statements}_i = f_i(\text{Economic Events}_i)
        \]
        \item Two firms have \textcolor{hkustred}{\textbf{comparable}} accounting systems if, for a \textbf{given set of economic events}, they produce \textbf{similar financial statements}.
        \item Comparability $\neq$ Uniformity: firms may use different accounting methods yet still produce comparable outputs.
    \end{itemize}
\end{frame}

\begin{frame}{Theoretical Mechanism: Why Comparability Benefits Analysts}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{itemize}
                \item \textbf{Better Benchmarks:} More comparable firms constitute better benchmarks for evaluating performance.
                \item \textbf{Information Transfer:} Financial information of comparable peers is more transferable \citep{ramnath2002investor, gleason2008contagion, durnev2009corporate}.
                \item \textbf{Lower Acquisition Cost:} Analysts face lower costs when acquiring and processing information about firms with comparable peers.
                \item \textbf{Net Effect:} A richer, higher-quality information set for the analyst.
            \end{itemize}
        \end{column}
        \begin{column}{0.42\textwidth}
            \centering
            \resizebox{\textwidth}{!}{

\section{Literature \& Contribution}

\begin{frame}{Prior Literature on Comparability}
    \begin{columns}[T]

            \begin{itemize}

            \vspace{0.2cm}
            \textcolor{hkustred}{\textbf{Limitations:}} Static, not firm-specific, and do not capture dynamic similarities over time.

\begin{frame}{Measuring Comparability: The Accounting Function}
    \textbf{Step 1:} Estimate each firm's accounting function using 16 quarters of data:
    \[
        Earnings_{it} = \alpha_i + \beta_i \, Return_{it} + \varepsilon_{it} \quad \text{--- Eq.\ (2)}
    \]
    \vspace{0.1cm}
    \textbf{Step 2:} Predict earnings using both firm $i$'s and firm $j$'s functions, holding economic events constant (using firm $i$'s returns):
    \begin{align*}
        E(Earnings)_{iit} &= \hat{\alpha}_i + \hat{\beta}_i \, Return_{it} \\
        E(Earnings)_{ijt} &= \hat{\alpha}_j + \hat{\beta}_j \, Return_{it}
    \end{align*}
    \textbf{Step 3:} Comparability = negative average absolute difference:
    \[
        CompAcct_{ijt} = -\frac{1}{16} \sum_{t-15}^{t} \bigl| E(Earnings_{iit}) - E(Earnings_{ijt}) \bigr|
    \]
    Greater (less negative) values $\Rightarrow$ \textcolor{hkustred}{\textbf{greater comparability}}.
\end{frame}

\begin{frame}{Aggregation to Firm-Level Measures}
    \begin{itemize}
        \item $CompAcct_{ijt}$ is computed for \textbf{every firm-pair} $(i,j)$ within the same 2-digit SIC industry.
        \item \textbf{Firm-level aggregation:}
        \begin{itemize}
            \item \textcolor{hkustred}{\textbf{CompAcct4$_{it}$}}: Average $CompAcct_{ijt}$ of the \textbf{four most comparable} firms $j$ to firm $i$.
            \item \textcolor{hkustred}{\textbf{CompAcctInd$_{it}$}}: Median $CompAcct_{ijt}$ for \textbf{all firms $j$} in the same industry as firm $i$.
        \end{itemize}
        \item Both measures are \textbf{dynamic} (updated annually), \textbf{firm-specific}, and \textbf{output-based}.
    \end{itemize}
\end{frame}

\begin{frame}{Why These Proxies?}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{\textcolor{hkustblue}{Stock Return $\to$ Economic Events}}
            \begin{itemize}
                \item Returns proxy for the \textbf{net effect} of economic events on the firm \citep{kothari2001capital}.
                \item Widely used in earnings--returns literature.
                \item Holding return constant across two firms isolates the \textbf{accounting function} difference.
            \end{itemize}
            \vspace{0.2cm}
            \textcolor{hkustred}{\textbf{Limitation:}} Prices may lead earnings $\Rightarrow$ addressed via \textit{CompAcct-PLE} (adding lagged return).
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{\textcolor{hkustblue}{Earnings $\to$ Financial Statements}}
            \begin{itemize}
                \item Earnings is the most important summary measure of income statement performance.
                \item Keeps the model \textbf{parsimonious and tractable}.
                \item Output-based: avoids ad hoc decisions about which accounting choices to examine.
            \end{itemize}
            \vspace{0.2cm}
            \textcolor{hkustred}{\textbf{Limitation:}} Captures only one financial statement dimension; balance-sheet comparability is not measured.
        \end{column}
    \end{columns}
\end{frame}

\section{Hypotheses}

\begin{frame}{Hypotheses and Logic}
    \begin{block}{H1: Analyst Coverage}
        Ceteris paribus, financial statement comparability is \textbf{positively associated} with analysts' coverage.
    \end{block}
    \textbf{Logic:} Lower information acquisition costs $\Rightarrow$ more analysts willing to cover the firm. Consistent with \citet{lang1996corporate}: analyst following increases in disclosure quality.
    
    \vspace{0.3cm}
    \begin{block}{H2a: Forecast Accuracy}
        Ceteris paribus, financial statement comparability is \textbf{positively associated} with analysts' forecast accuracy.
    \end{block}
    \textbf{Logic:} Higher-quality information from comparable peers helps analysts better evaluate historical performance and forecast earnings.
    
    \vspace{0.3cm}
    \begin{block}{H2b: Forecast Dispersion}
        Ceteris paribus, financial statement comparability is \textbf{negatively associated} with analysts' forecast dispersion.
    \end{block}
    \textbf{Logic:} Superior public information $\Rightarrow$ analysts place more weight on public signals, less on private information $\Rightarrow$ forecasts converge.
\end{frame}

\begin{frame}{Counter-Arguments (Null Hypotheses)}
    \begin{itemize}
        \item \textbf{H1 Null:} Better information environment may \textit{decrease} investor demand for analysts, thus fewer analysts cover the firm.
        \begin{itemize}
            \item \textcolor{gray}{But empirical evidence shows analysts primarily \textit{interpret} information rather than convey new information \citep{lang1996corporate}, so lower cost dominates.}
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{H2a Null:} If information from comparable firms is \textit{noisy or biased}, comparability could reduce accuracy.
        \vspace{0.2cm}
        \item \textbf{H2b Null:} Analysts may \textit{interpret} public information differently \citep{harris1993differences, kandel1995differential}, so more public information could \textit{increase} dispersion.
    \end{itemize}
\end{frame}

\section{Models \& Validity}

\begin{frame}{Model Specifications}
    \textbf{Analyst Coverage (OLS):}
    \[
        Coverage_{it+1} = \alpha + \beta_1 \, Comparability_{it} + \gamma \, Controls_{it} + \varepsilon_{it+1}  \quad \text{--- Eq.\ (10)}
    \]
    \textbf{Forecast Accuracy \& Dispersion (OLS):}
    \[
        Forecast\text{-}Metric_{it+1} = \alpha + \beta_1 \, Comparability_{it} + \gamma \, Controls_{it} + \varepsilon_{it+1} \quad \text{--- Eq.\ (11)}
    \]
    where $Forecast\text{-}Metric$ is either $Accuracy$ or $Dispersion$.
    \vspace{0.3cm}
    
    \textbf{Controls:} Size, Book-Market, Volume, R\&D, Depreciation, Issue, Predictability, Volatility\textsubscript{Earn}, Volatility\textsubscript{Ret}, SUE, Neg UE, Loss, Neg SI, Days.

    \vspace{0.2cm}
    \textbf{Estimation:} Industry \& year FE; standard errors clustered at firm and year levels \citep{petersen2009estimating}.
\end{frame}

\begin{frame}{Firm-Pair Level Tests}
    \textbf{Construct Validity --- Peer Selection (Probit):}
    \[
        AnalystComp_{ikj} = \alpha + \beta_1 \, CompAcct_{ij} + \gamma \, Controls_j + \varepsilon_{ikj} \quad \text{--- Eq.\ (6)}
    \]
    $AnalystComp_{ikj} = 1$ if analyst $k$ writing about firm $i$ refers to firm $j$ as a peer.
    
    \vspace{0.3cm}
    \textbf{Conditional Coverage (Probit):}
    \[
        CondCoverage_{ikjt} = \alpha + \beta_1 \, CompAcct_{ijt} + \gamma \, Controls_{jt} + \varepsilon_{ikjt} \quad \text{--- Eq.\ (7)}
    \]
    $CondCoverage_{ikjt} = 1$ if analyst $k$ covering firm $i$ also covers firm $j$.
    
    \vspace{0.3cm}
    \textbf{Correlated Forecast Errors (OLS):}
    \[
        CorrFcstError_{ijt} = -|FcstError_{it} - FcstError_{jt}| \quad \text{--- Eq.\ (8)}
    \]
    Higher values $\Rightarrow$ more systematic (correlated) errors across comparable firms.
\end{frame}

\begin{frame}{Construct Validity: Hand-Collected Analyst Reports}
    \begin{columns}[T]
        \begin{column}{0.55\textwidth}
            \begin{itemize}
                \item \textbf{Design:} Hand-collected 1,000 analyst reports from Investext for FY2005.
                \item Extracted firm $j$ peers mentioned by analysts when analyzing firm $i$.
                \item \textbf{Treatment:} $j$ chosen as peer ($AnalystComp=1$). \textbf{Control:} randomly matched firm in same 2-digit SIC.
                \item \textbf{Result:} $CompAcct_{ij}$ is \textbf{positively and significantly} associated with the probability of being selected as a peer.
                \item A 1-SD increase in $CompAcct$ $\Rightarrow$ 5\% increase in probability of being a peer (industry-matched); 3\% (industry-size-BM matched).
            \end{itemize}
        \end{column}
        \begin{column}{0.42\textwidth}
            \centering
            \resizebox{\textwidth}{!}{
            \begin{tikzpicture}[
                node distance=0.6cm,
                box/.style={draw, rounded corners, fill=gray!10, align=center, minimum width=3.5cm, minimum height=0.7cm, font=\small},

            ]

\begin{frame}{Descriptive Validation of the Measure}
    The measure behaves as expected across economic and earnings partitions:
    \vspace{0.2cm}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{\textcolor{hkustblue}{Economic Partitions}}
            \begin{itemize}
                \item \textbf{Same industry:} Mean $CompAcct = -2.7$
                \item \textbf{Cross industry:} Mean $\approx -4.1$
                \item \textbf{Same size quintile:} $-5.6$
                \item \textbf{Opposite size quintile:} $-6.7$
            \end{itemize}
        \end{column}
        \begin{column}{0.48\textwidth}
            \textbf{\textcolor{hkustblue}{Earnings Attribute Partitions}}
            \begin{itemize}
                \item Similar \textbf{accrual quality}: $-5.3$ vs.\ $-6.1$
                \item Similar \textbf{predictability}: $-4.8$ vs.\ $-5.1$
                \item Similar \textbf{smoothness}: $-4.4$ vs.\ $-5.3$
                \item Both \textbf{profit/loss}: $-3.7$ vs.\ $-7.1$
            \end{itemize}
        \end{column}
    \end{columns}
    \vspace{0.3cm}
    \begin{center}
        \textcolor{hkustred}{\textbf{Takeaway:}} Firms that are more economically similar and have more similar earnings attributes exhibit \textbf{higher comparability} --- consistent with the intended construct.
    \end{center}
\end{frame}

\begin{frame}{Endogeneity Concerns \& Robustness}
    \begin{itemize}
        \item \textbf{Omitted Variables:} Firm innovation could cause both low comparability and poor information environment.
        \begin{itemize}
            \item Controls for size, B/M, industry FE, volatility, and matching strategies mitigate this.
        \end{itemize}
        \item \textbf{Reverse Causality:} Analysts may pressure firms to adopt comparable methods \citep{jung2010demand}.
        \begin{itemize}
            \item Cannot fully rule out; acknowledged as a caveat.
        \end{itemize}
        \item \textbf{Prices Lead Earnings:} \textit{CompAcct-PLE} adds lagged returns to Eq.\ (2) --- results are robust.
        \item \textbf{Correlated Earnings:} \textit{CompAcct-R$^2$} uses the $R^2$ from regressing firm $i$'s earnings on firm $j$'s earnings --- confirms analyst coverage results; weaker for accuracy/dispersion.
        \item \textbf{Industry Definition:} Robust to 4-digit SIC and Fama--French 48 industries.
        \item \textbf{Size \& B/M Partitions:} Results hold for both large and small firms.
    \end{itemize}
\end{frame}


% ============================================================
% Conclusion
% ============================================================
\section{Summary}

\begin{frame}{Key Takeaways}
    \begin{enumerate}
        \item \textbf{Novel Measure:} Develops a firm-specific, output-based, dynamic measure of comparability using only widely available data.
        \item \textbf{Construct Validity:} Confirmed via hand-collected analyst reports and expected behavior across economic/earnings partitions.
        \item \textbf{Benefits to Analysts:}
        \begin{itemize}
            \item[$\checkmark$] Coverage: $\uparrow$ 10\% for a 1-SD increase in CompAcct4.
            \item[$\checkmark$] Accuracy: $\uparrow$ 23\% improvement for a 1-SD increase.
            \item[$\checkmark$] Dispersion: $\downarrow$ 27\% reduction for a 1-SD increase.
        \end{itemize}
        \item \textbf{Broader Implication:} The measure can assess comparability changes from new standards (e.g., IFRS adoption).
    \end{enumerate}
    \vspace{0.3cm}
    \begin{center}
        \textcolor{hkustred}{\textbf{Bottom line:}} Comparability lowers information acquisition costs and improves the quality of analysts' information sets.
    \end{center}
\end{frame}
