\section{Introduction}

% --- SLIDE 1: Motivation + Prior Literature + Gap ---
\begin{frame}{Motivation, Prior Literature \& Gap}
    \begin{itemize}
        \item \textbf{Importance:} The SEC \citeyearpar{sec2000} and FASB \citeyearpar{fasb1980} stress that comparable financial statements facilitate efficient capital allocation. Textbooks universally emphasize cross-firm comparability.
        \item \textbf{Prior Literature:}
        \begin{itemize}
            \item \textit{Input-based:} \citet{bradshaw2008harmonizing}, \citet{defond2003empirical}, \citet{bradshaw2009accounting} --- examine accounting method heterogeneity (e.g., LIFO vs.\ FIFO). Requires ad hoc decisions on which choices to use and costly data collection.
            \item \textit{Cross-sectional levels:} \citet{joos1994effects}, \citet{land2002empirical} --- compare contemporaneous levels (ROE, multiples) at a single point. Static, not firm-specific.
        \end{itemize}
        \item \textbf{Gap:} Despite widespread agreement on importance, \textcolor{hkustred}{\textbf{no empirical construct}} of comparability existed, and \textcolor{hkustred}{\textbf{little evidence}} documented its benefits to users.
    \end{itemize}
\end{frame}


% --- SLIDE 2: Research Questions ---
\begin{frame}{Research Questions}
    \vspace{0.5cm}
    \begin{center}
        \Large \textit{``How can we \textbf{measure} financial statement comparability, and does higher comparability \textbf{benefit} financial statement users?''}
    \end{center}
    \vspace{0.5cm}
    \begin{itemize}
        \item \textbf{Develop} a firm-specific, \textcolor{hkustred}{\textbf{output-based}}, quantitative measure of financial statement comparability --- contrast with input-based and static approaches in prior work.
        \item \textbf{Document} benefits to sell-side analysts: coverage, forecast accuracy, and forecast dispersion.
    \end{itemize}
\end{frame}

\section{Comparability Measure}

% --- SLIDE 3: Definition of Comparability ---
\begin{frame}{Definition of Comparability}
    \begin{block}{FASB Definition}
        ``Comparability is the quality of information that enables users to identify \textbf{similarities and differences} between two sets of economic phenomena'' \citep{fasb1980}.
    \end{block}
    \vspace{0.3cm}
    \begin{itemize}
        \item The \textbf{accounting system} is a mapping from economic events to financial statements:
        \[
            \text{Financial Statements}_i = f_i(\text{Economic Events}_i)
        \]
        \item Two firms have \textcolor{hkustred}{\textbf{comparable}} accounting systems if, for a \textbf{given set of economic events}, they produce \textbf{similar financial statements}.
        \item Comparability $\neq$ Uniformity: firms may use different accounting methods yet still produce comparable outputs.
    \end{itemize}
\end{frame}

% --- SLIDE 5: Measuring Comparability ---
\begin{frame}{Measuring Comparability: The Accounting Function}
    \textbf{Step 1:} Estimate each firm's accounting function using 16 quarters of data:
    \[
        Earnings_{it} = \alpha_i + \beta_i \, Return_{it} + \varepsilon_{it} \quad \text{--- Eq.\ (2)}
    \]
    \vspace{0.1cm}
    \textbf{Step 2:} Predict earnings using both firm $i$'s and firm $j$'s functions, holding economic events constant (using firm $i$'s returns):
    \begin{align*}
        E(Earnings)_{iit} &= \hat{\alpha}_i + \hat{\beta}_i \, Return_{it} \\
        E(Earnings)_{ijt} &= \hat{\alpha}_j + \hat{\beta}_j \, Return_{it}
    \end{align*}
    \textbf{Step 3:} Comparability = negative average absolute difference:
    \[
        CompAcct_{ijt} = -\frac{1}{16} \sum_{t-15}^{t} \bigl| E(Earnings_{iit}) - E(Earnings_{ijt}) \bigr|
    \]
    Greater (less negative) values $\Rightarrow$ \textcolor{hkustred}{\textbf{greater comparability}}.
\end{frame}

% --- SLIDE 6: Aggregation to Firm Level ---
\begin{frame}{Aggregation to Firm-Level Measures}
    \begin{itemize}
        \item $CompAcct_{ijt}$ is computed for \textbf{every firm-pair} $(i,j)$ within the same 2-digit SIC industry.
        \item \textbf{Firm-level aggregation:}
        \begin{itemize}
            \item \textcolor{hkustred}{\textbf{CompAcct4$_{it}$}}: Average $CompAcct_{ijt}$ of the \textbf{four most comparable} firms $j$ to firm $i$.
            \item \textcolor{hkustred}{\textbf{CompAcctInd$_{it}$}}: Median $CompAcct_{ijt}$ for \textbf{all firms $j$} in the same industry as firm $i$.
        \end{itemize}
        \item Both measures are \textbf{dynamic} (updated annually), \textbf{firm-specific}, and \textbf{output-based}.
    \end{itemize}
    \vspace{0.3cm}
    \textbf{Why these proxies?}
    \begin{itemize}
        \item \textbf{Stock return} proxies for economic events \citep{kothari2001capital}; holding it constant isolates the accounting function difference.
        \item \textbf{Earnings} is the key income statement output --- keeps the model parsimonious and avoids ad hoc choices about which accounting methods to examine.
    \end{itemize}
\end{frame}

\section{Hypotheses}

% --- SLIDE 7: Hypotheses and Logic ---
\begin{frame}{Hypotheses and Logic}
    \small
    \begin{block}{H1: Analyst Coverage}
        \small Financial statement comparability is \textbf{positively associated} with analysts' coverage.
    \end{block}
    \vspace{-0.1cm}
    \small\textbf{Logic:} Lower info acquisition costs $\Rightarrow$ more analysts cover the firm \citep{bhushan1989analyst, lang1996corporate}.
    
    \vspace{0.1cm}
    \begin{block}{H2a: Forecast Accuracy}
        \small Financial statement comparability is \textbf{positively associated} with analysts' forecast accuracy.
    \end{block}
    \vspace{-0.1cm}
    \small\textbf{Logic:} Higher-quality information from comparable peers helps analysts better evaluate performance and forecast earnings.
    
    \vspace{0.1cm}
    \begin{block}{H2b: Forecast Dispersion}
        \small Financial statement comparability is \textbf{negatively associated} with analysts' forecast dispersion.
    \end{block}
    \vspace{-0.1cm}
    \small\textbf{Logic:} Superior public info $\Rightarrow$ more weight on public signals, less on private $\Rightarrow$ forecasts converge.
\end{frame}

\section{Models \& Validity}

% --- SLIDE 8: Model Specifications ---
\begin{frame}{Model Specifications}
    \textbf{Analyst Coverage (OLS, Eq.\ 10):}
    \[
        Coverage_{it+1} = \alpha + \beta_1 \, Comparability_{it} + \gamma \, Controls_{it} + \varepsilon_{it+1}
    \]

    \textbf{Forecast Accuracy \& Dispersion (OLS, Eq.\ 11):}
    \[
        Forecast\text{-}Metric_{it+1} = \alpha + \beta_1 \, Comparability_{it} + \gamma \, Controls_{it} + \varepsilon_{it+1}
    \]
    where $Forecast\text{-}Metric \in \{Accuracy, \; Dispersion\}$.
    \vspace{0.3cm}
    
    \textbf{Controls:}
    \begin{itemize}
        \item Eq.\ (10): Size, B/M, Volume, R\&D, Depreciation, Issue, Predictability, Volatility\textsubscript{Earn}, Volatility\textsubscript{Ret}.
        \item Eq.\ (11): adds SUE, Neg UE, Loss, Neg SI, Days.
    \end{itemize}

    \vspace{0.2cm}
    \textbf{Estimation:} Industry \& year FE; SE clustered at firm and year levels \citep{petersen2009estimating}.
\end{frame}
